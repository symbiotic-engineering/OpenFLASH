\documentclass{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[scale=0.9]

% -----------------------
% Parameters
% -----------------------
\def\aone{1.0}
\pgfmathsetmacro{\atwo}{2*\aone}
\def\dtwo{1.0}
\pgfmathsetmacro{\done}{3*\dtwo}
\pgfmathsetmacro{\hdepth}{6}

% -----------------------
% Free surface (TOP)
% -----------------------
\draw[thick] (-4,0) -- (4,0);
\draw[cyan, ultra thick, smooth, samples=200, domain=-4:-2]
plot (\x,{2/3*sin(deg(2*pi*\x/2))});
\draw[cyan, ultra thick, smooth, samples=200, domain=2:4]
plot (\x,{-2/3*sin(deg(2*pi*\x/2))});

% -----------------------
% Cylinders (hanging downward)
% -----------------------

% Outer shell portion (0 to d2)
\fill[gray!25]
(-\atwo,0) --
(-\atwo,-\dtwo) --
(-\aone,-\dtwo) --
(-\aone,-\done) --
(\aone,-\done) --
(\aone,-\dtwo) --
(\atwo,-\dtwo) --
(\atwo,0) -- cycle;

% Body outline (external boundary only)

\draw[thick] (-\atwo,0) -- (\atwo,0);          % top
\draw[thick] (-\atwo,0) -- (-\atwo,-\dtwo);
\draw[thick] (\atwo,0) -- (\atwo,-\dtwo);
\draw[thick] (-\atwo,-\dtwo) -- (-\aone,-\dtwo);
\draw[thick] (\aone,-\dtwo) -- (\atwo,-\dtwo);
\draw[thick] (-\aone,-\dtwo) -- (-\aone,-\done);
\draw[thick] (\aone,-\dtwo) -- (\aone,-\done);
\draw[thick] (-\aone,-\done) -- (\aone,-\done);

% -----------------------
% Seabed
% -----------------------
\draw[thick] (-4,-\hdepth) -- (4,-\hdepth);

% -----------------------
% Dashed region boundaries
% -----------------------

% Vertical continuation below outer cylinder
\draw[dashed] (-\atwo,-\dtwo) -- (-\atwo,-\hdepth);
\draw[dashed] (\atwo,-\dtwo) -- (\atwo,-\hdepth);

% Vertical continuation below inner cylinder
\draw[dashed] (-\aone,-\done) -- (-\aone,-\hdepth);
\draw[dashed] (\aone,-\done) -- (\aone,-\hdepth);

% middle r=0 line
\draw[dashed] (0,0) -- (0,-\hdepth);

% -----------------------
% Red ovals around dashed lines
% -----------------------
\draw[red, ultra thick] (-\atwo,-3.5) ellipse (0.25 and 2.2);
\draw[red, ultra thick] ( \atwo,-3.5) ellipse (0.25 and 2.2);

\draw[red, ultra thick] (-\aone,-4.5) ellipse (0.25 and 1.3);
\draw[red, ultra thick] ( \aone,-4.5) ellipse (0.25 and 1.3);

% -----------------------  arrow for heave
\draw[<->, ultra thick] (0,.5) -- (0,-.5);

\end{tikzpicture}
\end{document}
